program teste4
declare
integer divisor, number;
boolean nofactor, prime
do read(number);
write(number);
divisor := number;
nofactor := true;
while nofactor and (divisor < 1) do
if (number mod divisor) = 0 then nofactor := false
else divisor := divisor - 1
end
end;
prime := nofactor;
if prime then write('S') else write('N') end
end
